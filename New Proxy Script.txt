function SurgFinish(varlist, packet)

if AUTO_SURG_TOGGLE == true then
        if varlist[2]:find("Hey, somebody left") or varlist[2]:find("The patient's parents are so glad you saved their child") or varlist[2]:find("After a surgery like that") or varlist[2]:find("The Growtopian Hippocratic Society has awarded you") then
            
            stopTime = os.time()
            local elapsedTime = stopTime - startTime
            SendPacket(2, "action|input\ntext|`c[`8@CENSORED Autosurg`c]`2 You have completed a surgery! Time it took: " .. elapsedTime .. " seconds")
            os.execute("cls")

            
        
        end
    end

		
	end
end
